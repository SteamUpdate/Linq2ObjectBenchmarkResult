# `Array.FindAll(q => q...)` (Изменённое условие)

public Entity[] Common()
{
    return array.Where(q => q.SomeInt % 2 == 0).ToArray();
}

public Entity[] Optimized()
{
    return Array.FindAll(array, q => q.SomeInt % 2 == 0);
}
